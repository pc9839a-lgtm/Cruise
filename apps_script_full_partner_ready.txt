/* =========================
 * GET : bootstrap
 * ========================= */
function doGet(e) {
  try {
    const params = (e && e.parameter) ? e.parameter : {};
    const action = toText_(params.action);

    if (action === 'blog_index') {
      return createJsonOrJsonpOutput_(
        getBlogIndexPayloadResponse_(),
        params.callback
      );
    }

    if (action === 'bootstrap' || action === 'bootstrap_full') {
      return createJsonOrJsonpOutput_(
        getBootstrapPayloadResponse_('full'),
        params.callback
      );
    }

    if (action === 'bootstrap_initial') {
      return createJsonOrJsonpOutput_(
        getBootstrapPayloadResponse_('initial'),
        params.callback
      );
    }

    if (action === 'bootstrap_deferred') {
      return createJsonOrJsonpOutput_(
        getBootstrapPayloadResponse_('deferred'),
        params.callback
      );
    }

    if (action === 'bootstrap_cache_warm') {
      const warmed = buildAndCacheBootstrapPayloads_();
      return createJsonOrJsonpOutput_({
        success: true,
        initial_keys: Object.keys(warmed.initial),
        deferred_keys: Object.keys(warmed.deferred)
      }, params.callback);
    }

    if (action === 'agent_signup') {
      return createJsonOrJsonpOutput_(
        getAgentSignupPayloadResponse_(params),
        params.callback
      );
    }

    return createJsonOrJsonpOutput_({
      success: false,
      message: '지원하지 않는 요청입니다.'
    }, params.callback);
  } catch (error) {
    return createJsonOrJsonpOutput_({
      success: false,
      message: String(error && error.message ? error.message : error)
    }, (e && e.parameter) ? e.parameter.callback : '');
  }
}

/* =========================
 * POST : inquiry submit
 * ========================= */
function doPost(e) {
  try {
    const params = (e && e.parameter) ? e.parameter : {};
    validateInquiry_(params);

    const settings = getSettingsObjectCached_();
    const defaultEmail = toText_(settings.default_email);
    if (!defaultEmail) {
      throw new Error('설정 시트의 default_email 값이 비어 있습니다.');
    }

    const agents = getActiveAgentsCached_();
    const agentCode = toText_(params.agent_code);
    const matchedAgent = findAgent_(agents, agentCode);

    const recipientPool = [];

    if (matchedAgent && toText_(matchedAgent.email)) {
      recipientPool.push(toText_(matchedAgent.email));
    }

    if (defaultEmail) {
      recipientPool.push(defaultEmail);
    }

    const normalizedRecipients = normalizeEmailList_(recipientPool.join(','));
    const assignedEmail = normalizedRecipients.join(',');

    const createdAt = Utilities.formatDate(
      new Date(),
      Session.getScriptTimeZone() || 'Asia/Seoul',
      'yyyy-MM-dd HH:mm:ss'
    );

    const inquiryData = {
      created_at: createdAt,
      name: toText_(params.name),
      phone: toText_(params.phone).replace(/\D+/g, ''),
      interest_schedule_id: firstNonEmpty_(params.interest_schedule_id, params.schedule_id),
      people_count: toText_(params.people_count),
      region_detail: toText_(params.region_detail),
      travel_ready_status: toText_(params.travel_ready_status),
      message: firstNonEmpty_(params.message, params.memo),
      agent_code: agentCode,
      utm_source: toText_(params.utm_source),
      utm_medium: toText_(params.utm_medium),
      utm_campaign: toText_(params.utm_campaign),
      page_url: firstNonEmpty_(params.page_url, params.landing_page),
      referrer: toText_(params.referrer),
      assigned_email: assignedEmail,
      mail_sent: 'N',
      status: '신규'
    };

    const isPartnerInquiry = isPartnerInquiry_(params);
    const targetSheetName = isPartnerInquiry
      ? SHEET_NAMES.partnerInquiries
      : SHEET_NAMES.inquiries;

    appendByHeader_(
      targetSheetName,
      isPartnerInquiry
        ? buildPartnerInquiryData_({
            params: params,
            inquiryData: inquiryData,
            createdAt: createdAt,
            agentCode: agentCode,
            assignedEmail: assignedEmail,
            settings: settings
          })
        : inquiryData
    );

    const mailResult = sendInquiryEmail_({
      createdAt: createdAt,
      params: inquiryData,
      receiverEmail: assignedEmail,
      siteTitle: firstNonEmpty_(settings.site_title, settings.site_name, '크루즈 일정 문의'),
      agentName: matchedAgent ? firstNonEmpty_(matchedAgent.agent_name, matchedAgent.name, matchedAgent.agent_code) : ''
    });

    try {
      appendByHeader_(SHEET_NAMES.mailLogs, {
        sent_at: createdAt,
        name: inquiryData.name,
        phone: inquiryData.phone,
        agent_code: inquiryData.agent_code,
        receiver_email: assignedEmail,
        result: mailResult.success ? 'SUCCESS' : 'FAIL',
        error_message: mailResult.success ? '' : mailResult.message
      });
    } catch (logError) {
      console.error('알림로그 저장 실패:', logError);
    }

    return createJsonOrJsonpOutput_({
      success: true,
      message: mailResult.success
        ? '문의가 정상 접수되었습니다.'
        : '문의는 저장되었지만 메일 발송은 실패했습니다.'
    });
  } catch (error) {
    console.error(error);
    return createJsonOrJsonpOutput_({
      success: false,
      message: String(error && error.message ? error.message : error)
    });
  }
}

/* =========================
 * Settings
 * ========================= */
function getSettingsObject_(store) {
  const internalStore = store || createSheetStore_([SHEET_NAMES.settings]);
  const entry = getStoredSheet_(internalStore, SHEET_NAMES.settings);
  const values = entry ? entry.values : [];

  if (!values.length) return {};

  const result = {};

  for (let i = 1; i < values.length; i += 1) {
    const key = toText_(values[i][0]);
    const value = toText_(values[i][1]);
    if (key) result[key] = value;
  }

  return result;
}

/* =========================
 * Agents
 * ========================= */
function getActiveAgents_(store) {
  const internalStore = store || createSheetStore_([SHEET_NAMES.agents]);
  const entry = getStoredSheet_(internalStore, SHEET_NAMES.agents);
  if (!entry) return [];

  const rows = entry.rows;
  return rows
    .map(function (row) {
      return {
        agent_code: firstNonEmpty_(row.agent_code, row.code),
        agent_name: firstNonEmpty_(row.agent_name, row.name),
        email: firstNonEmpty_(row.email, row.receiver_email),
        signup_url: firstNonEmpty_(row.signup_url, row.membership_url),
        is_active: firstNonEmpty_(row.is_active, row.use_yn, row.display_yn, row.active, 'Y')
      };
    })
    .filter(function (item) {
      return toText_(item.agent_code) && toText_(item.email) && isDisplayOn_(item.is_active);
    });
}

function findAgent_(agents, agentCode) {
  const code = toText_(agentCode);
  if (!code) return null;

  for (let i = 0; i < agents.length; i += 1) {
    if (toText_(agents[i].agent_code) === code) {
      return agents[i];
    }
  }
  return null;
}

function getAgentSignupPayloadResponse_(params) {
  const agentCode = toText_(params && params.agent);
  const agents = getActiveAgentsCached_();
  const matchedAgent = findAgent_(agents, agentCode);

  return {
    success: !!(matchedAgent && toText_(matchedAgent.signup_url)),
    agent_code: matchedAgent ? toText_(matchedAgent.agent_code) : '',
    agent_name: matchedAgent ? toText_(matchedAgent.agent_name) : '',
    signup_url: matchedAgent ? toText_(matchedAgent.signup_url) : ''
  };
}

/* =========================
 * Schedule list
 * ========================= */
function getSchedules_(store) {
  const internalStore = store || createSheetStore_([SHEET_NAMES.schedules]);
  const entry = getStoredSheet_(internalStore, SHEET_NAMES.schedules);
  if (!entry) return [];

  const rows = entry.rows;

  return rows
    .map(function (row, index) {
      const copied = {};
      Object.keys(row || {}).forEach(function (key) {
        copied[key] = row[key];
      });
      copied.__row_index = index;
      return copied;
    })
    .filter(function (row) {
      return toText_(row.schedule_id) && toText_(row.title) && isDisplayOn_(firstNonEmpty_(row.display_yn, 'Y'));
    })
    .sort(function (a, b) {
      return compareSortKeepingSheetOrder_(a.sort_no, b.sort_no, a.__row_index, b.__row_index);
    })
    .map(function (row) {
      return {
        schedule_id: toText_(row.schedule_id),
        title: toText_(row.title),
        region: toText_(row.region),
        ship_name: toText_(row.ship_name),
        departure_date: formatDateCell_(row.departure_date),
        return_date: formatDateCell_(row.return_date),
        departure_airport: toText_(row.departure_airport),
        summary: toText_(row.summary),
        thumbnail_url: toText_(row.thumbnail_url),
        schedule_image_url: toText_(row.schedule_image_url),
        detail_intro: toText_(row.detail_intro),
        route_ports: toText_(row.route_ports),
        highlight_yn: toText_(row.highlight_yn)
      };
    });
}

/* =========================
 * Schedule days
 * ========================= */
function getScheduleDays_(store) {
  const internalStore = store || createSheetStore_([SHEET_NAMES.scheduleDays]);
  const entry = getStoredSheet_(internalStore, SHEET_NAMES.scheduleDays);
  if (!entry) return [];

  const rows = entry.rows;

  return rows
    .filter(function (row) {
      return toText_(row.schedule_id) && String(firstNonEmpty_(row.day_no, '')).trim();
    })
    .sort(function (a, b) {
      const scheduleCompare = toText_(a.schedule_id).localeCompare(toText_(b.schedule_id), 'ko');
      if (scheduleCompare !== 0) return scheduleCompare;
      return Number(firstNonEmpty_(a.day_no, 0)) - Number(firstNonEmpty_(b.day_no, 0));
    })
    .map(function (row) {
      return {
        schedule_id: toText_(row.schedule_id),
        day_no: String(firstNonEmpty_(row.day_no, '')),
        date: formatDateCell_(row.date),
        port_name: toText_(row.port_name),
        arrival_time: normalizeTimeCell_(row.arrival_time),
        departure_time: normalizeTimeCell_(row.departure_time),
        description: toText_(row.description),
        image_url: toText_(row.image_url)
      };
    });
}

/* =========================
 * Bootstrap / cache / store
 * ========================= */
function getBootstrapPayloadResponse_(mode) {
  const normalizedMode = normalizeBootstrapMode_(mode);

  if (normalizedMode === 'full') {
    return getBootstrapFullPayloadResponse_();
  }

  const cacheKey = getBootstrapCacheKey_(normalizedMode);
  const cached = getJsonCache_(cacheKey);
  if (cached && hasUsableBootstrapPayload_(normalizedMode, cached)) {
    return cached;
  }

  const lastSuccess = getJsonCache_(getLastSuccessCacheKey_(cacheKey));

  try {
    const builtPayload = buildBootstrapPayload_(normalizedMode);
    const payload = mergePayloadWithLastSuccess_(builtPayload, lastSuccess);

    if (hasUsableBootstrapPayload_(normalizedMode, payload)) {
      putJsonCacheWithLastSuccess_(cacheKey, payload, CACHE_TTL_SECONDS[cacheKey] || 300);
      return payload;
    }

    if (lastSuccess) return lastSuccess;
    return payload;
  } catch (error) {
    if (lastSuccess) return lastSuccess;
    throw error;
  }
}

function getBootstrapFullPayloadResponse_() {
  const initialKey = getBootstrapCacheKey_('initial');
  const deferredKey = getBootstrapCacheKey_('deferred');

  const cachedInitial = getJsonCache_(initialKey);
  const cachedDeferred = getJsonCache_(deferredKey);

  if (
    cachedInitial &&
    cachedDeferred &&
    hasUsableBootstrapPayload_('initial', cachedInitial) &&
    hasUsableBootstrapPayload_('deferred', cachedDeferred)
  ) {
    return mergeBootstrapPayloads_(cachedInitial, cachedDeferred);
  }

  const lastInitial = getJsonCache_(getLastSuccessCacheKey_(initialKey));
  const lastDeferred = getJsonCache_(getLastSuccessCacheKey_(deferredKey));

  try {
    const payloads = buildAndCacheBootstrapPayloads_();
    const initialPayload = mergePayloadWithLastSuccess_(payloads.initial, lastInitial);
    const deferredPayload = mergePayloadWithLastSuccess_(payloads.deferred, lastDeferred);

    if (
      hasUsableBootstrapPayload_('initial', initialPayload) ||
      hasUsableBootstrapPayload_('deferred', deferredPayload)
    ) {
      return mergeBootstrapPayloads_(initialPayload, deferredPayload);
    }

    if (lastInitial || lastDeferred) {
      return mergeBootstrapPayloads_(lastInitial || {}, lastDeferred || {});
    }

    return mergeBootstrapPayloads_(initialPayload, deferredPayload);
  } catch (error) {
    if (lastInitial || lastDeferred) {
      return mergeBootstrapPayloads_(lastInitial || {}, lastDeferred || {});
    }
    throw error;
  }
}

function getBlogIndexPayloadResponse_() {
  const cacheKey = 'blog_index';
  const cached = getJsonCache_(cacheKey);
  if (cached && Array.isArray(cached.posts) && cached.posts.length > 0) {
    return cached;
  }

  const lastSuccess = getJsonCache_(getLastSuccessCacheKey_(cacheKey));

  try {
    const store = createSheetStore_([SHEET_NAMES.contentLinks]);
    const posts = getContentLinks_(store);

    if (!posts.length && lastSuccess) {
      return lastSuccess;
    }

    const categories = buildBlogIndexCategories_(posts);
    const payload = {
      success: true,
      post_count: posts.length,
      categories: categories,
      posts: posts
    };

    if (posts.length) {
      putJsonCacheWithLastSuccess_(
        cacheKey,
        payload,
        CACHE_TTL_SECONDS.blog_index
      );
    }

    return payload;
  } catch (error) {
    if (lastSuccess) return lastSuccess;
    throw error;
  }
}

function buildBlogIndexCategories_(posts) {
  const seen = {};
  const categories = [];

  (posts || []).forEach(function (post) {
    const category = toText_(post && post.category);
    if (!category || seen[category]) return;
    seen[category] = true;
    categories.push(category);
  });

  return categories;
}

function buildAndCacheBootstrapPayloads_() {
  const store = createSheetStore_(BOOTSTRAP_ALL_SHEETS);
  const initialPayload = buildBootstrapInitialPayloadFromStore_(store);
  const deferredPayload = buildBootstrapDeferredPayloadFromStore_(store);

  const initialKey = getBootstrapCacheKey_('initial');
  const deferredKey = getBootstrapCacheKey_('deferred');

  const lastInitial = getJsonCache_(getLastSuccessCacheKey_(initialKey));
  const lastDeferred = getJsonCache_(getLastSuccessCacheKey_(deferredKey));

  const safeInitialPayload = mergePayloadWithLastSuccess_(initialPayload, lastInitial);
  const safeDeferredPayload = mergePayloadWithLastSuccess_(deferredPayload, lastDeferred);

  if (hasUsableBootstrapPayload_('initial', safeInitialPayload)) {
    putJsonCacheWithLastSuccess_(
      initialKey,
      safeInitialPayload,
      CACHE_TTL_SECONDS.bootstrap_initial
    );
  }

  if (hasUsableBootstrapPayload_('deferred', safeDeferredPayload)) {
    putJsonCacheWithLastSuccess_(
      deferredKey,
      safeDeferredPayload,
      CACHE_TTL_SECONDS.bootstrap_deferred
    );
  }

  return {
    initial: safeInitialPayload,
    deferred: safeDeferredPayload
  };
}

function buildBootstrapPayload_(mode) {
  const normalizedMode = normalizeBootstrapMode_(mode);
  const sheetNames = normalizedMode === 'initial'
    ? BOOTSTRAP_INITIAL_SHEETS
    : BOOTSTRAP_DEFERRED_SHEETS;
  const store = createSheetStore_(sheetNames);

  if (normalizedMode === 'initial') {
    return buildBootstrapInitialPayloadFromStore_(store);
  }

  return buildBootstrapDeferredPayloadFromStore_(store);
}

function buildBootstrapInitialPayloadFromStore_(store) {
  return {
    settings: getSettingsObject_(store),
    schedules: getSchedules_(store),
    basic_info: getBasicInfo_(store)
  };
}

function buildBootstrapDeferredPayloadFromStore_(store) {
  return {
    schedule_days: getScheduleDays_(store),
    reviews: getReviews_(store),
    targets: getTargets_(store),
    process_steps: getProcessSteps_(store),
    cabins: getCabins_(store),
    faqs: getFaqs_(store),
    trust_points: getTrustPoints_(store),
    content_links: getContentLinks_(store)
  };
}

function mergeBootstrapPayloads_(initialPayload, deferredPayload) {
  const merged = {};
  Object.keys(initialPayload || {}).forEach(function (key) {
    merged[key] = initialPayload[key];
  });
  Object.keys(deferredPayload || {}).forEach(function (key) {
    merged[key] = deferredPayload[key];
  });
  return merged;
}

function normalizeBootstrapMode_(mode) {
  const text = toText_(mode).toLowerCase();
  if (text === 'initial') return 'initial';
  if (text === 'deferred') return 'deferred';
  return 'full';
}

function getBootstrapCacheKey_(mode) {
  const normalizedMode = normalizeBootstrapMode_(mode);
  return normalizedMode === 'initial'
    ? 'bootstrap_initial'
    : normalizedMode === 'deferred'
      ? 'bootstrap_deferred'
      : 'bootstrap_full';
}

function getSettingsObjectCached_() {
  return getOrBuildCachedJson_('settings', CACHE_TTL_SECONDS.settings, function () {
    return getSettingsObject_();
  });

function getActiveAgentsCached_() {

function getOrBuildCachedJson_(cacheName, ttlSeconds, builder) {
  const cached = getJsonCache_(cacheName);
  if (cached) return cached;

  const lastSuccess = getJsonCache_(getLastSuccessCacheKey_(cacheName));

  try {
    const payload = builder();

    if (shouldCacheGenericPayload_(cacheName, payload)) {
      putJsonCacheWithLastSuccess_(cacheName, payload, ttlSeconds);
      return payload;
    }

    if (lastSuccess) return lastSuccess;
    return payload;
  } catch (error) {
    if (lastSuccess) return lastSuccess;
    throw error;
  }
}

function putJsonCacheWithLastSuccess_(cacheName, payload, ttlSeconds) {
  if (!shouldCacheGenericPayload_(cacheName, payload)) return;

  putJsonCache_(cacheName, payload, ttlSeconds);
  putJsonCache_(getLastSuccessCacheKey_(cacheName), payload, LAST_SUCCESS_TTL_SECONDS);
}

function shouldCacheGenericPayload_(cacheName, payload) {
  if (!payload) return false;

  if (cacheName === 'settings') {
    return Object.keys(payload || {}).length > 0;
  }

  if (cacheName === 'agents') {
    return Array.isArray(payload) && payload.length > 0;
  }

  if (cacheName === 'blog_index') {
    return Array.isArray(payload.posts) && payload.posts.length > 0;
  }

  if (cacheName === 'bootstrap_initial') {
    return hasUsableBootstrapPayload_('initial', payload);
  }

  if (cacheName === 'bootstrap_deferred') {
    return hasUsableBootstrapPayload_('deferred', payload);
  }

  if (cacheName === 'bootstrap_full') {
    return hasUsableBootstrapPayload_('full', payload);
  }

  return true;
}

function hasUsableBootstrapPayload_(mode, payload) {
  if (!payload) return false;

  const normalizedMode = normalizeBootstrapMode_(mode);

  if (normalizedMode === 'initial') {
    return (
      isNonEmptyArray_(payload.schedules) ||
      isNonEmptyArray_(payload.basic_info)
    );
  }

  if (normalizedMode === 'deferred') {
    return (
      isNonEmptyArray_(payload.schedule_days) ||
      isNonEmptyArray_(payload.reviews) ||
      isNonEmptyArray_(payload.targets) ||
      isNonEmptyArray_(payload.process_steps) ||
      isNonEmptyArray_(payload.cabins) ||
      isNonEmptyArray_(payload.faqs) ||
      isNonEmptyArray_(payload.trust_points) ||
      isNonEmptyArray_(payload.content_links)
    );
  }

  return (
    isNonEmptyArray_(payload.schedules) ||
    isNonEmptyArray_(payload.basic_info) ||
    isNonEmptyArray_(payload.schedule_days) ||
    isNonEmptyArray_(payload.reviews) ||
    isNonEmptyArray_(payload.targets) ||
    isNonEmptyArray_(payload.process_steps) ||
    isNonEmptyArray_(payload.cabins) ||
    isNonEmptyArray_(payload.faqs) ||
    isNonEmptyArray_(payload.trust_points) ||
    isNonEmptyArray_(payload.content_links)
  );
}

function mergePayloadWithLastSuccess_(payload, lastSuccess) {
  if (!lastSuccess) return payload || {};

  const merged = {};

  Object.keys(payload || {}).forEach(function (key) {
    const value = payload[key];
    const lastValue = lastSuccess[key];

    if (
      Array.isArray(value) &&
      value.length === 0 &&
      Array.isArray(lastValue) &&
      lastValue.length > 0
    ) {
      merged[key] = lastValue;
      return;
    }

    if (
      key === 'settings' &&
      value &&
      typeof value === 'object' &&
      !Array.isArray(value) &&
      Object.keys(value).length === 0 &&
      lastValue &&
      typeof lastValue === 'object' &&
      !Array.isArray(lastValue) &&
      Object.keys(lastValue).length > 0
    ) {
      merged[key] = lastValue;
      return;
    }

    merged[key] = value;
  });

  Object.keys(lastSuccess || {}).forEach(function (key) {
    if (!Object.prototype.hasOwnProperty.call(merged, key)) {
      merged[key] = lastSuccess[key];
    }
  });

  return merged;
}

function isNonEmptyArray_(value) {
  return Array.isArray(value) && value.length > 0;
}

function getLastSuccessCacheKey_(cacheName) {
  return cacheName + '_last_success';
}

function getJsonCache_(cacheName) {
  const cache = CacheService.getScriptCache();
  const raw = cache.get(buildCacheKey_(cacheName));
  if (!raw) return null;

  try {
    return JSON.parse(raw);
  } catch (error) {
    return null;
  }
}

function putJsonCache_(cacheName, payload, ttlSeconds) {
  try {
    CacheService
      .getScriptCache()
      .put(
        buildCacheKey_(cacheName),
        JSON.stringify(payload),
        Math.max(1, Number(ttlSeconds) || 300)
      );
  } catch (error) {
    // CacheService 용량 제한에 걸리면 조용히 통과
  }
}

function buildCacheKey_(cacheName) {
  return CACHE_PREFIX + ':' + toText_(cacheName);
}

function createSheetStore_(sheetNames) {
  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const uniqueSheetNames = (sheetNames || []).filter(function (name, index, arr) {
    return !!name && arr.indexOf(name) === index;
  });

  const store = {};

  uniqueSheetNames.forEach(function (sheetName) {
    const sheet = spreadsheet.getSheetByName(sheetName);

    if (!sheet) {
      store[sheetName] = null;
      return;
    }

    const values = sheet.getDataRange().getValues();
    const headers = values.length
      ? values[0].map(function (header) { return toText_(header); })
      : [];
    const rows = values.length > 1
      ? values.slice(1).map(function (row) {
          const item = {};
          headers.forEach(function (header, idx) {
            item[header] = row[idx];
          });
          return item;
        })
      : [];

    store[sheetName] = {
      sheet: sheet,
      values: values,
      headers: headers,
      rows: rows
    };
  });

  return store;
}

function getStoredSheet_(store, sheetName) {
  if (!store || !Object.prototype.hasOwnProperty.call(store, sheetName)) {
    return null;
  }
  return store[sheetName];
}

function validateInquiry_(params) {
  if (!toText_(params.name)) {
    throw new Error('이름이 없습니다.');
  }

  if (!toText_(params.phone).replace(/\D+/g, '')) {
    throw new Error('연락처가 없습니다.');
  }

  if (!firstNonEmpty_(params.interest_schedule_id, params.schedule_id)) {
    throw new Error('관심 일정이 없습니다.');
  }

  if (!toText_(params.people_count)) {
    throw new Error('인원수가 없습니다.');
  }
}

function appendByHeader_(sheetName, data) {
  const sheet = getSheetOrThrow_(sheetName);
  const headers = getHeaderRow_(sheet);

  if (!headers.length) {
    throw new Error(sheetName + ' 시트의 헤더가 없습니다.');
  }

  const row = headers.map(function (header) {
    return Object.prototype.hasOwnProperty.call(data, header) ? data[header] : '';
  });

  sheet.appendRow(row);
}

function createJsonOrJsonpOutput_(payload, callback) {
  const safeCallback = toText_(callback);

  if (safeCallback) {
    if (!/^[a-zA-Z0-9_.$]+$/.test(safeCallback)) {
      throw new Error('callback 형식이 올바르지 않습니다.');
    }

    return ContentService
      .createTextOutput(safeCallback + '(' + JSON.stringify(payload) + ');')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }

  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}

function getSheetOrThrow_(sheetName) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet) {
    throw new Error(sheetName + ' 시트를 찾을 수 없습니다.');
  }
  return sheet;
}

function getHeaderRow_(sheet) {
  const lastColumn = sheet.getLastColumn();
  if (lastColumn < 1) return [];
  return sheet.getRange(1, 1, 1, lastColumn).getValues()[0].map(function (header) {
    return toText_(header);
  });
}

function getRowsByHeader_(sheet) {
  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];

  const headers = values[0].map(function (header) {
    return toText_(header);
  });

  return values.slice(1).map(function (row) {
    const item = {};
    headers.forEach(function (header, idx) {
      item[header] = row[idx];
    });
    return item;
  });
}

/* =========================
 * Value helpers
 * ========================= */
function toText_(value) {
  if (value === null || value === undefined) return '';
  return String(value).trim();
}

function firstNonEmpty_() {
  for (var i = 0; i < arguments.length; i += 1) {
    var value = toText_(arguments[i]);
    if (value) return value;
  }
  return '';
}

function isDisplayOn_(value) {
  const text = toText_(value).toUpperCase();
  if (!text) return true;
  return !['N', 'NO', 'FALSE', '0', 'HIDE', 'HIDDEN'].includes(text);
}

function compareSort_(aSort, bSort, aTitle, bTitle) {
  const a = parseSortNo_(aSort);
  const b = parseSortNo_(bSort);

  if (a !== b) return a - b;
  return toText_(aTitle).localeCompare(toText_(bTitle), 'ko');
}

function compareSortKeepingSheetOrder_(aSort, bSort, aRowIndex, bRowIndex) {
  const aHasSort = toText_(aSort) !== '';
  const bHasSort = toText_(bSort) !== '';

  if (aHasSort && bHasSort) {
    const aNum = parseSortNo_(aSort);
    const bNum = parseSortNo_(bSort);
    if (aNum !== bNum) return aNum - bNum;
  } else if (aHasSort && !bHasSort) {
    return -1;
  } else if (!aHasSort && bHasSort) {
    return 1;
  }

  return Number(aRowIndex || 0) - Number(bRowIndex || 0);
}

function parseSortNo_(value) {
  const num = Number(value);
  return isNaN(num) ? 999999 : num;
}

function formatDateCell_(value) {
  if (!value && value !== 0) return '';

  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, Session.getScriptTimeZone() || 'Asia/Seoul', 'yyyy-MM-dd');
  }

  const text = toText_(value);
  if (!text) return '';

  const parsed = new Date(text);
  if (!isNaN(parsed.getTime()) && /^\d{4}[-/]\d{1,2}[-/]\d{1,2}/.test(text)) {
    return Utilities.formatDate(parsed, Session.getScriptTimeZone() || 'Asia/Seoul', 'yyyy-MM-dd');
  }

  return text;
}

function normalizeTimeCell_(value) {
  if (value === null || value === undefined) return '';

  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, Session.getScriptTimeZone() || 'Asia/Seoul', 'H:mm');
  }

  return toText_(value);
}

function normalizeEmailList_(value) {
  return String(value || '')
    .split(',')
    .map(function (email) {
      return String(email || '').trim();
    })
    .filter(function (email) {
      return !!email;
    })
    .filter(function (email, index, arr) {
      return arr.indexOf(email) === index;
    });
}
